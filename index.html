Sambungan & Konfigurasi Google Sheets

Sistem ini menyokong penyimpanan secara automatik ke awan Google Sheets. Sila tampalkan URL Web App Google Sheets anda di bawah. Pautan ini akan disimpan secara kekal di dalam memori pelayar dan tidak akan hilang walaupun selepas halaman dimuat semula (*refresh*).

Panduan Pemutus & Penyegerakan Google Sheets

Bagi memastikan fungsi pemadaman dipautkan ke Google Sheet pada ruangan baris yang betul (*backward loop*), sila pastikan kod Web App di bawah disalin sepenuhnya ke **Extensions > Apps Script** di Google Sheet anda, kemudian deploy semula sebagai Web App baharu.

Kod Apps Script Code.gs (Versi Diperbetulkan): 
function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    var requestData = JSON.parse(e.postData.contents);
    var action = requestData.action;
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheets()[0]; 
    
    if (action === 'delete') {
      var id = requestData.id;
      var values = sheet.getDataRange().getValues();
      var deleted = false;
      // Gelung ke belakang dari baris bawah ke atas (backward loop) bagi mencegah ralat anjakan baris / index shifting
      for (var i = values.length - 1; i >= 1; i--) {
        if (values[i][0] == id) { 
          sheet.deleteRow(i + 1);
          deleted = true;
        }
      }
      return ContentService.createTextOutput(JSON.stringify({ success: true, deleted: deleted })).setMimeType(ContentService.MimeType.JSON);
    }
    
    // Simpan Rekod Baharu
    sheet.appendRow([
      requestData.id,
      requestData.nama,
      requestData.bidang,
      requestData.komponen,
      requestData.tarikh,
      requestData.tp,
      requestData.eviden,
      requestData.fileName || ""
    ]);
    return ContentService.createTextOutput(JSON.stringify({ success: true })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}
